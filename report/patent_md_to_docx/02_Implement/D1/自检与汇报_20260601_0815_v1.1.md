```powershell
$psi = New-Object System.Diagnostics.ProcessStartInfo
$psi.FileName = "C:/Users/98514/.conda/envs/roo_mcp/python.exe"
$psi.Arguments = "D:/FJL/Projects/Kilo_Code_Gobal_Settings/Roo_Code_utils/server.py"
$psi.UseShellExecute = $false
$psi.RedirectStandardOutput = $true
$psi.RedirectStandardError = $true
$psi.RedirectStandardInput = $true
$psi.EnvironmentVariables["PYTHONIOENCODING"] = "utf-8"
$psi.EnvironmentVariables["PYTHONUTF8"] = "1"
$p = [System.Diagnostics.Process]::Start($psi)
Start-Sleep -Seconds 2
if ($p.HasExited) {
    Write-Host "启动失败: exit code $($p.ExitCode)"
} else {
    Write-Host "server.py 运行正常 (pid: $($p.Id))"
    $p.Kill()
}
```

**结果**:

```
server.py 运行正常 (pid: 44952)
```
