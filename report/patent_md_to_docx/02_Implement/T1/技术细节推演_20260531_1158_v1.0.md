## 2. 核心实现策略：启发式分类器

### 2.1 为什么放弃 AST heading token

目标 MD 不使用 `# heading` 语法。其"标题"以以下形式出现：
- `权 利 要 求 书` — 空格分隔的纯文本（节标题）
- `**技术领域**` — 粗体独占行（子标题）
- `基于空间侵占反馈的机器人导航策略训练方法及系统` — 纯文本独立行（发明名称）

这些在 markdown-it-py AST 中全部被解析为 `paragraph_open → inline → paragraph_close`，无一例外。因此无法依赖 `heading_open` token。

### 2.2 `classify_section()` 分类器实现

```python
import re

def classify_section(text: str) -> str:
    """
    对段落文本进行分类，返回语义标签。
    返回: 'section_title' | 'sub_title' | 'claim' | 'numbered_para' | 'normal'
    """
    stripped = text.strip()
    
    # 1. 权利要求行: "N. 正文..." (行首数字+点+空格)
    if re.match(r'^\d+\.\s', stripped):
        return 'claim'
    
    # 2. 说明书编号段落: "[NNNN] 正文..." (行首方括号4位数字)
    if re.match(r'^\[(\d{4})\](?:\s*\*\*\s*\*\*)?', stripped):
        return 'numbered_para'
    
    # 3. 节标题: 连续空格分隔的中文字符串 + 总长 < 30字符
    #    如 "权 利 要 求 书"、"说 明 书 摘 要"
    if re.match(r'^[\u4e00-\u9fff](?:\s[\u4e00-\u9fff])+$', stripped) and len(stripped) < 30:
        return 'section_title'
    
    # 4. 粗体子标题: **XXX** 独占整行
    if re.match(r'^\*\*[^*]+\*\*$', stripped):
        return 'sub_title'
    
    return 'normal'
```

### 2.3 迭代增强路径

当未来 MD 采用 MyST 格式（支持 `# heading` 和 `$$` 公式）时，分类器升级为：

```python
# 优先检测 heading token
if token.type == 'heading_open':
    level = int(token.tag[1])  # 'h1' → 1
    return f'heading_{level}'

# 回退到启发式分类
return classify_section(inline_text)
```

## 4. 字体引擎实现 (Sprint 1 核心交付物)

```python
from docx.oxml.ns import qn
from docx.shared import Pt

def set_run_font(run, cn_font='宋体', en_font='Times New Roman',
                 size=Pt(12), bold=False):
    """
    一步到位设置中英文混排字体 + 字号 + 加粗。
    包含 v2.0 审查通过的 w:hint='eastAsia' 防御性渲染提示。
    """
    run.font.name = en_font
    run.font.size = size
    run.font.bold = bold

    rPr = run._element.get_or_add_rPr()
    rFonts = rPr.find(qn('w:rFonts'))
    if rFonts is None:
        rFonts = rPr.makeelement(qn('w:rFonts'), {})
        rPr.insert(0, rFonts)

    rFonts.set(qn('w:eastAsia'), cn_font)
    rFonts.set(qn('w:ascii'), en_font)
    rFonts.set(qn('w:hAnsi'), en_font)
    rFonts.set(qn('w:hint'), 'eastAsia')  # 阻断等线回退


def init_patent_document() -> Document:
    """创建符合 CNIPA 规范的空白专利文档"""
    from docx import Document
    from docx.shared import Cm

    doc = Document()

    # 全局默认字体
    style = doc.styles['Normal']
    style.font.name = 'Times New Roman'
    style.font.size = Pt(12)
    style.element.rPr.rFonts.set(qn('w:eastAsia'), '宋体')

    # CNIPA 页边距: 左25 上25 右15 下15
    for section in doc.sections:
        section.top_margin = Cm(2.5)
        section.bottom_margin = Cm(1.5)
        section.left_margin = Cm(2.5)
        section.right_margin = Cm(1.5)

    return doc
```

## 5. 主循环骨架 (Sprint 2 核心交付物)

```python
import markdown_it
import re

def build_patent_docx(md_text: str) -> Document:
    """专利 MD → CNIPA 合规 DOCX 主入口"""
    md = markdown_it.MarkdownIt('commonmark', {'breaks': False})
    tokens = md.parse(md_text)
    doc = init_patent_document()

    i = 0
    while i < len(tokens):
        token = tokens[i]

        if token.type == 'paragraph_open':
            inline = tokens[i + 1]          # inline token
            raw_text = inline.content        # 原始段落文本
            tag = classify_section(raw_text)

            if tag == 'section_title':
                add_section_title(doc, raw_text)
            elif tag == 'sub_title':
                inner = re.sub(r'^\*\*|\*\*$', '', raw_text).strip()
                add_sub_title(doc, inner)
            elif tag == 'numbered_para':
                m = re.match(r'^\[(\d{4})\]\s*\*\*\s*\*\*\s*(.*)', raw_text)
                if m:
                    add_body_paragraph(doc, m.group(2), number=m.group(1))
                else:
                    add_body_paragraph(doc, raw_text)
            elif tag == 'claim':
                m = re.match(r'^(\d+)\.\s(.+)', raw_text, re.DOTALL)
                if m:
                    add_claim_paragraph(doc, m.group(1), m.group(2))
            else:
                # 处理行内元素
                add_rich_paragraph(doc, inline)

            i += 3  # 跳过 paragraph_open, inline, paragraph_close

        elif token.type == 'table_open':
            i = _parse_table_block(doc, tokens, i)

        elif token.type == 'hr':
            doc.add_page_break()
            i += 1

        elif token.type == 'heading_open':
            level = int(token.tag[1])
            text = tokens[i + 1].content
            p = doc.add_heading(text, level=level)
            i += 3

        else:
            i += 1

    return doc


def add_rich_paragraph(doc, inline_token) -> Paragraph:
    """处理包含行内元素的段落 (bold/italic/text)"""
    p = doc.add_paragraph()
    p.paragraph_format.first_line_indent = Pt(24)
    p.paragraph_format.line_spacing = 1.5

    children = inline_token.children if inline_token.children else []
    for child in children:
        if child.type == 'strong_open':
            continue  # nesting=1, 不生成 Run
        elif child.type == 'strong_close':
            continue  # nesting=-1, 不生成 Run
        elif child.type == 'em_open':
            continue
        elif child.type == 'em_close':
            continue
        elif child.type == 'text':
            txt = child.content
            if not txt.strip():
                continue  # 跳过纯空白 (过滤 ** ** 污染)
            # 检查前一个 token 是否为 strong_open
            is_bold = (children[children.index(child) - 1].type == 'strong_open'
                       if children.index(child) > 0 else False)
            is_italic = (children[children.index(child) - 1].type == 'em_open'
                         if children.index(child) > 0 else False)
            run = p.add_run(txt)
            run.bold = is_bold
            run.italic = is_italic
            set_run_font(run, size=Pt(12))

    return p
```

## 6. 已知风险与防御策略

| 风险项 | 严重度 | 防御策略 |
|--------|:---:|---------|
| `classify_section` 误分类 | P2 | 所有分类失败均降级为 `normal` 段落，不影响文档可读性 |
| `** **` bold spaces 污染 | P2 | `add_rich_paragraph` 中 `not txt.strip()` 跳过纯空白 Run |
| 图片路径断裂 | P1 | 多级回退: 严格匹配 → 相对 MD 目录解析 → `*_md/figures/` 同名猜测 → 灰色占位 |
| 表格嵌套 strong 丢失 | P2 | 表格子解析器需独立遍历 td inline children |
| markdown-it-py 中文兼容 | P1 | 2.2.0 版本经大量中文项目 (Jupyter Book/MyST) 验证，风险可控 |
| CNIPA 2026 XML 政策 | P2 | DOCX 仍为事务所内部 Draft 流转标准格式，本工具定位不受影响 |
